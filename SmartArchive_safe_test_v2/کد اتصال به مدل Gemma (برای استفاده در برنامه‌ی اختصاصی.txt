"""
اتصال به مدل Gemma 3 4B با استفاده از Ollama API
"""

import requests
import json
import base64
from pathlib import Path

class GemmaConnector:
    def __init__(self, model_name="gemma3:4b", base_url="http://localhost:11434"):
        self.model_name = model_name
        self.base_url = base_url
        self.api_url = f"{base_url}/api/chat"

    def chat(self, prompt: str, system_prompt: str = None, max_tokens: int = 1024) -> str:
        """ارسال پیام به مدل Gemma و دریافت پاسخ"""
        messages = []
        if system_prompt:
            messages.append({"role": "system", "content": system_prompt})
        messages.append({"role": "user", "content": prompt})
        
        payload = {
            "model": self.model_name,
            "messages": messages,
            "stream": False,
            "options": {"num_predict": max_tokens}
        }
        
        try:
            response = requests.post(self.api_url, json=payload, timeout=60)
            if response.status_code == 200:
                return response.json()["message"]["content"]
            else:
                return f"❌ خطا: {response.status_code} - {response.text}"
        except Exception as e:
            return f"❌ خطا در اتصال به Ollama: {e}"

    def analyze_image(self, image_path: str, question: str = "این تصویر چه چیزی را نشان می‌دهد؟") -> str:
        """تحلیل تصویر با Gemma (در صورت پشتیبانی)"""
        # Gemma 3 4B از vision پشتیبانی می‌کند
        try:
            with open(image_path, 'rb') as f:
                image_base64 = base64.b64encode(f.read()).decode('utf-8')
            
            # تعیین نوع تصویر
            ext = Path(image_path).suffix.lower()
            mime_type = {
                '.jpg': 'image/jpeg', '.jpeg': 'image/jpeg',
                '.png': 'image/png', '.gif': 'image/gif',
                '.webp': 'image/webp'
            }.get(ext, 'image/jpeg')
            
            payload = {
                "model": self.model_name,
                "messages": [
                    {
                        "role": "user",
                        "content": question,
                        "images": [image_base64]
                    }
                ],
                "stream": False
            }
            
            response = requests.post(self.api_url, json=payload, timeout=120)
            if response.status_code == 200:
                return response.json()["message"]["content"]
            else:
                return f"❌ خطا: {response.status_code}"
        except Exception as e:
            return f"❌ خطا در تحلیل تصویر: {e}"

    def classify_image(self, image_path: str) -> Tuple[str, str]:
        """دسته‌بندی تصویر با Gemma"""
        prompt = """
        این تصویر را تحلیل کن و دسته‌بندی کن.
        دسته‌های ممکن: انسان, حیوان, منظره, گیاه, غذا, ساختمان, خودرو, اشیاء, لوگو, سایر
        فقط نام دسته را برگردان.
        اگر دسته فرعی هم تشخیص دادی، به شکل "دسته اصلی / دسته فرعی" برگردان.
        """
        response = self.analyze_image(image_path, prompt)
        # پردازش پاسخ برای استخراج دسته‌بندی
        categories = ["انسان", "حیوان", "منظره", "گیاه", "غذا", "ساختمان", "خودرو", "اشیاء", "لوگو", "سایر"]
        for cat in categories:
            if cat in response:
                return (cat, "")
        return ("سایر", "")
