class DatabaseManager:
    def __init__(self, db_path="archivepro.db"):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.cursor = self.conn.cursor()
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS files (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                source_path TEXT,
                dest_path TEXT,
                category TEXT,
                sub_category TEXT,
                filename TEXT,
                file_hash TEXT,
                size INTEGER,
                date_taken TEXT,
                metadata TEXT,
                scan_date TEXT
            )
        ''')
        self.conn.commit()

    def insert_file(self, data):
        self.cursor.execute('''
            INSERT OR REPLACE INTO files (
                source_path, dest_path, category, sub_category,
                filename, file_hash, size, date_taken, metadata, scan_date
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            data.get('source', ''),
            data.get('dest', ''),
            data.get('category', ''),
            data.get('sub_cat', ''),
            Path(data.get('dest', '')).name,
            data.get('metadata', {}).get('file_hash', ''),
            data.get('metadata', {}).get('size', 0),
            data.get('metadata', {}).get('date_taken', ''),
            json.dumps(data.get('metadata', {})),
            datetime.now().isoformat()
        ))
        self.conn.commit()

    def get_all_files(self):
        self.cursor.execute('SELECT * FROM files ORDER BY filename')
        rows = self.cursor.fetchall()
        cols = [d[0] for d in self.cursor.description]
        return [dict(zip(cols, row)) for row in rows]

    def close(self):
        if self.conn:
            self.conn.close()

class ContentDetector:
    @staticmethod
    def detect(image_path: str) -> Tuple[str, str]:
        filename = Path(image_path).stem.lower()
        
        # کلمات کلیدی برای دسته‌بندی
        if any(k in filename for k in ['person', 'people', 'man', 'woman', 'child', 'انسان', 'مرد', 'زن']):
            return ("انسان", "")
        if any(k in filename for k in ['food', 'meal', 'pizza', 'pasta', 'burger', 'kebab', 'غذا', 'خوراک']):
            return ("غذا", "")
        if any(k in filename for k in ['cat', 'dog', 'horse', 'lion', 'tiger', 'گربه', 'سگ', 'شیر']):
            return ("حیوان", "")
        if any(k in filename for k in ['sea', 'ocean', 'beach', 'mountain', 'river', 'lake', 'forest', 'دریا', 'کوه', 'جنگل']):
            return ("منظره", "")
        if any(k in filename for k in ['flower', 'rose', 'tulip', 'tree', 'گل', 'رز', 'درخت']):
            return ("گیاه", "")
        if any(k in filename for k in ['building', 'house', 'church', 'mosque', 'temple', 'ساختمان', 'مسجد', 'کلیسا']):
            return ("ساختمان", "")
        if any(k in filename for k in ['car', 'automobile', 'motorcycle', 'truck', 'ماشین', 'خودرو']):
            return ("خودرو", "")
        if any(k in filename for k in ['logo', 'brand', 'لوگو', 'برند']):
            return ("اشیاء", "لوگو")
        if any(k in filename for k in ['plate', 'cup', 'bowl', 'glass', 'ظرف', 'بشقاب', 'لیوان']):
            return ("اشیاء", "ظروف_آشپزخانه")
        
        return ("سایر", "")

class ClassificationEngine:
    EXTENSION_MAP = {
        'وکتور': ['.cdr', '.ai', '.eps', '.svg', '.psd', '.afdesign'],
        'معماری': ['.dwg', '.dxf', '.skp', '.rvt', '.ifc', '.step', '.stp'],
        'سه‌بعدی': ['.blend', '.max', '.3ds', '.c4d', '.fbx', '.obj', '.stl'],
        'موسیقی': ['.mp3', '.wav', '.flac', '.aac', '.ogg', '.m4a', '.wma'],
        'ویدئو': ['.mp4', '.avi', '.mkv', '.mov', '.wmv', '.flv', '.webm'],
        'اسناد': ['.pdf', '.txt', '.rtf', '.md', '.log', '.csv', '.json', '.xml', '.html', '.css', '.js', '.py', '.cpp', '.java', '.php', '.sql'],
        'آفیس': ['.doc', '.docx', '.xls', '.xlsx', '.ppt', '.pptx'],
        'نرم‌افزار': ['.exe', '.msi', '.apk', '.deb', '.dmg'],
        'فشرده': ['.zip', '.rar', '.7z', '.tar', '.gz', '.iso']
    }

    def __init__(self, src, dst, delete_source=False):
        self.src = Path(src)
        self.dst = Path(dst)
        self.delete_source = delete_source
        self.detector = ContentDetector()
        self.db = DatabaseManager()
        self.paused = False
        self.stopped = False
        self.total_files = 0

    def set_pause(self, paused): self.paused = paused
    def set_stop(self, stopped): self.stopped = stopped

    def _calculate_hash(self, path):
        hasher = hashlib.md5()
        try:
            with open(path, 'rb') as f:
                while chunk := f.read(8192):
                    hasher.update(chunk)
            return hasher.hexdigest()
        except:
            return ""

    def _get_category_by_extension(self, file_path: Path) -> Tuple[str, str]:
        ext = file_path.suffix.lower()
        for category, exts in self.EXTENSION_MAP.items():
            if ext in exts:
                return (category, "سایر")
        return ("سایر", "")

    def _get_category(self, file_path: Path) -> Tuple[str, str]:
        ext = file_path.suffix.lower()
        if ext in ['.jpg', '.jpeg', '.png', '.gif', '.bmp', '.webp', '.ico', '.raw', '.heic']:
            return self.detector.detect(str(file_path))
        return self._get_category_by_extension(file_path)

    def _get_destination_path(self, file_path: Path, category: str, sub_category: str) -> Path:
        base = self.dst
        
        # تصاویر
        if category in ["انسان", "حیوان", "منظره", "گیاه", "غذا", "ساختمان", "خودرو", "اشیاء", "سایر"]:
            img_base = base / "تصاویر"
            if category != "سایر":
                if sub_category:
                    return img_base / category / sub_category / file_path.name
                return img_base / category / file_path.name
            return img_base / "سایر" / file_path.name
        
        # وکتور
        elif category == "وکتور":
            sw_map = {'.cdr': 'CorelDRAW', '.ai': 'Illustrator', '.psd': 'Photoshop', '.eps': 'EPS', '.svg': 'SVG'}
            sw = sw_map.get(file_path.suffix.lower(), "سایر")
            return base / "وکتور" / sw / file_path.name
        
        # معماری
        elif category == "معماری":
            sw_map = {'.dwg': 'AutoCAD', '.skp': 'SketchUp', '.rvt': 'Revit'}
            sw = sw_map.get(file_path.suffix.lower(), "سایر")
            return base / "معماری" / sw / file_path.name
        
        # سه‌بعدی
        elif category == "سه‌بعدی":
            sw_map = {'.blend': 'Blender', '.max': '3ds_Max', '.c4d': 'Cinema4D'}
            sw = sw_map.get(file_path.suffix.lower(), "سایر")
            return base / "سه‌بعدی" / sw / file_path.name
        
        # سایر
        else:
            return base / category / file_path.name

    def classify_and_copy(self, progress_callback=None, log_callback=None, pause_check_callback=None) -> Dict:
        files = list(self.src.rglob('*'))
        files = [f for f in files if f.is_file()]
        self.total_files = len(files)
        stats = {'total': self.total_files, 'processed': 0, 'failed': 0, 'skipped': 0, 'duplicates': 0}
        hash_map = {}

        for idx, file_path in enumerate(files):
            if pause_check_callback:
                while self.paused:
                    time.sleep(0.5)
                    if self.stopped:
                        break
                if self.stopped:
                    break

            try:
                fhash = self._calculate_hash(file_path)
                if not fhash:
                    stats['failed'] += 1
                    continue

                if fhash in hash_map:
                    stats['duplicates'] += 1
                    if log_callback:
                        log_callback(idx+1, self.total_files, f"🔄 تکراری: {file_path.name}", "")
                    continue

                main_cat, sub_cat = self._get_category(file_path)
                dest_path = self._get_destination_path(file_path, main_cat, sub_cat)

                if dest_path:
                    os.makedirs(dest_path.parent, exist_ok=True)
                    shutil.copy2(file_path, dest_path)
                    stats['processed'] += 1
                    hash_map[fhash] = file_path.name
                    self.db.insert_file({
                        'source': str(file_path),
                        'dest': str(dest_path),
                        'category': main_cat,
                        'sub_cat': sub_cat,
                        'metadata': {'file_hash': fhash, 'size': file_path.stat().st_size}
                    })
                    if log_callback:
                        sub_info = f" [{sub_cat}]" if sub_cat else ""
                        log_callback(idx+1, self.total_files, f"✅ {main_cat}{sub_info} → {file_path.name}", f"📁 {dest_path}")
                else:
                    stats['skipped'] += 1

            except Exception as e:
                stats['failed'] += 1
                if log_callback:
                    log_callback(idx+1, self.total_files, f"❌ خطا: {file_path.name}", str(e))

            if progress_callback:
                progress_callback(idx+1, self.total_files)

        if not self.stopped and self.delete_source:
            try:
                shutil.rmtree(self.src)
                if log_callback:
                    log_callback(self.total_files, self.total_files, "🗑️ پوشه مبدا حذف شد.", "")
            except:
                pass

        return stats

class ClassificationWorker(QThread):
    progress = Signal(int, int)
    log = Signal(str)
    finished = Signal(dict)

    def __init__(self, src, dst, delete):
        super().__init__()
        self.src = src
        self.dst = dst
        self.delete = delete
        self.paused = False
        self.stopped = False
        self.engine = None

    def toggle_pause(self):
        self.paused = not self.paused
        if self.engine:
            self.engine.set_pause(self.paused)

    def stop(self):
        self.stopped = True
        if self.engine:
            self.engine.set_stop(True)

    def run(self):
        self.engine = ClassificationEngine(self.src, self.dst, self.delete)
        stats = self.engine.classify_and_copy(
            progress_callback=lambda c, t: self.progress.emit(c, t),
            log_callback=lambda c, t, s, d: self.log.emit(s),
            pause_check_callback=lambda: None
        )
        self.finished.emit(stats)

    def update_texts(self):
        tr = self.tr
        self.setWindowTitle(tr.tr("app_title"))
        self.lbl_source.setText(tr.tr("source"))
        self.lbl_dest.setText(tr.tr("dest"))
        self.src_edit.setPlaceholderText(tr.tr("source_placeholder"))
        self.dst_edit.setPlaceholderText(tr.tr("dest_placeholder"))
        self.src_btn.setText(tr.tr("browse"))
        self.dst_btn.setText(tr.tr("browse"))
        self.del_cb.setText(tr.tr("delete"))
        self.start_btn.setText(tr.tr("start"))
        self.pause_btn.setText(tr.tr("pause"))
        self.stop_btn.setText(tr.tr("stop"))
        self.report_btn.setText(tr.tr("report"))
        self.progress.setFormat(tr.tr("progress"))
        self.status_label.setText(tr.tr("ready"))
        self.tabs.setTabText(0, tr.tr("tab_log"))
        self.tabs.setTabText(1, tr.tr("tab_report"))

    def show_about(self):
        QMessageBox.about(self, self.tr.tr("about"), self.tr.tr("about_text"))

    def start_classification(self):
        src = self.src_edit.text().strip()
        dst = self.dst_edit.text().strip()
        if not src or not dst:
            QMessageBox.warning(self, self.tr.tr("error"), self.tr.tr("error_folders"))
            return
        if src == dst:
            QMessageBox.warning(self, self.tr.tr("error"), self.tr.tr("error_same"))
            return
        if self.del_cb.isChecked():
            if QMessageBox.question(self, self.tr.tr("warning_title"), self.tr.tr("warning_text"), QMessageBox.Yes|QMessageBox.No) == QMessageBox.No:
                return

        self.start_btn.setEnabled(False)
        self.pause_btn.setEnabled(True)
        self.stop_btn.setEnabled(True)
        self.report_btn.setEnabled(False)
        self.log_list.clear()
        self.progress.setValue(0)
        self.status_label.setText(self.tr.tr("processing"))

        self.worker = ClassificationWorker(src, dst, self.del_cb.isChecked())
        self.worker.progress.connect(self.update_progress)
        self.worker.log.connect(self.add_log)
        self.worker.finished.connect(self.classification_finished)
        self.worker.start()

    def toggle_pause(self):
        if self.worker:
            self.is_paused = not self.is_paused
            self.worker.toggle_pause()
            self.pause_btn.setText(self.tr.tr("resume") if self.is_paused else self.tr.tr("pause"))
            self.status_label.setText(self.tr.tr("paused") if self.is_paused else self.tr.tr("processing"))

    def stop_classification(self):
        if self.worker:
            self.is_stopped = True
            self.worker.stop()
            self.stop_btn.setEnabled(False)
            self.status_label.setText("⏹️ Stopping...")

    def update_progress(self, current, total):
        self.progress.setMaximum(total)
        self.progress.setValue(current)
        self.count_label.setText(self.tr.tr("files", current, total))

    def add_log(self, msg):
        self.log_list.addItem(msg)
        self.log_list.scrollToBottom()

    def classification_finished(self, stats):
        self.start_btn.setEnabled(True)
        self.pause_btn.setEnabled(False)
        self.stop_btn.setEnabled(False)
        self.report_btn.setEnabled(True)
        self.status_label.setText(self.tr.tr("completed"))
        self.progress.setValue(stats['total'])
        self.report_txt.setText(f"Total: {stats['total']}\nCopied: {stats['processed']}\nFailed: {stats['failed']}")
        QMessageBox.information(self, self.tr.tr("info_done"), self.tr.tr("info_text", stats['processed']))

    def generate_report(self):
        dst = self.dst_edit.text().strip()
        if not dst:
            return
        data = self.db.get_all_files()
        if not data:
            return
        try:
            import csv
            with open(os.path.join(dst, "report.csv"), 'w', newline='', encoding='utf-8-sig') as f:
                writer = csv.writer(f)
                writer.writerow(["Filename", "Source", "Destination", "Category", "Sub Category"])
                for row in data:
                    writer.writerow([
                        Path(row.get('dest_path', '')).name,
                        row.get('source_path', ''),
                        row.get('dest_path', ''),
                        row.get('category', ''),
                        row.get('sub_category', '')
                    ])
            QMessageBox.information(self, "Report", f"Report saved in {dst}")
        except Exception as e:
            QMessageBox.warning(self, "Error", str(e))
